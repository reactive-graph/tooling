19mod embedded_resources {

27#[derive(Component)]
28pub struct GraphQlClientWebResourceProvider {}
29
30#[async_trait]
31#[component_alias]
32impl WebResourceProvider for GraphQlClientWebResourceProvider {
33    fn id(&self) -> Uuid {
34        *ID
35    }
36
37    fn get_context_path(&self) -> String {
38        CONTEXT_PATH.to_string()
39    }
40
41    async fn handle_web_resource(&self, path: String, _request: Request<HttpBody>) -> http::Result<Response<HttpBody>> {
42        let path = match path.as_str() {
43            "" | "index.html" | "graph.html" | "graph" => String::from("graph.html"),
44            "dynamic-graph.html" | "dynamic-graph" => String::from("dynamic-graph.html"),
45            _ => path,
46        };
47        let asset = embedded_resources::GraphQlClientWebResourceAsset::get(path.as_ref());
48        match asset {
49            Some(asset) => {
50                let body: HttpBody = match asset.data {
51                    Cow::Borrowed(bytes) => HttpBody::Binary(bytes.to_vec()),
52                    Cow::Owned(bytes) => HttpBody::Binary(bytes.to_vec()),
53                };
54                let mime_type = from_path(path.as_str()).first_or_octet_stream();
55                Response::builder()
56                    .status(StatusCode::OK)
57                    .header(CONTENT_TYPE, mime_type.to_string())
58                    .body(body)
59            }
60            None => Response::builder().status(StatusCode::NOT_FOUND).body(HttpBody::None),
61        }
62    }
63}
